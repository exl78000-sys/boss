Firebase Firestore 建議規則（配合 v34 歸入帳號）

注意：未登入資料不做裝置權限控管；只要不是 Google 帳號擁有的角色，任何未登入使用者輸入角色名稱即可修改。
Google 登入建立的資料仍只允許本人或管理者修改。已登入者可將未登入建立、尚未歸入帳號的角色資料更新並歸入自己的 Google 帳號。

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }
    function isAdmin() {
      return signedIn() && request.auth.token.email == 'exl78000@gmail.com';
    }
    function isGoogleOwned() {
      return resource.data.ownerUid != null && resource.data.ownerUid != '';
    }
    function isOwner() {
      return signedIn() && (
        resource.data.ownerUid == request.auth.uid ||
        resource.data.createdBy.uid == request.auth.uid
      );
    }
    function isCreatingGoogleOwned() {
      return signedIn() && (
        request.resource.data.ownerUid == request.auth.uid ||
        request.resource.data.createdBy.uid == request.auth.uid
      );
    }

    match /signups/{signupId} {
      allow read: if true;
      // Google 登入者建立自己的資料；未登入者也可建立無 ownerUid 的資料。
      allow create: if isAdmin() || isCreatingGoogleOwned() || !signedIn();
      // Google 登入建立的資料：只有本人或管理者可改。
      // 未登入建立的資料：不設裝置權限，任何人輸入角色名稱即可修改；已登入者更新時可歸入自己的 Google 帳號。
      allow update, delete: if isAdmin() || isOwner() || !isGoogleOwned();
    }

    match /users/{userId}/characters/{characterId} {
      allow read, write: if signedIn() && (request.auth.uid == userId || isAdmin());
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
