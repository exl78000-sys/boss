Firebase Firestore 建議規則（v48 手動角色綁定修正版）

重點：
- 送出報名不會自動綁定 Google 帳號。
- 只有按「綁定到我的 Google 帳號」才會把角色主檔 characters/{角色名稱} 綁定到目前帳號。
- 權限以 characters 角色主檔 ownerUid 為準。
- signups 報名資料不再作為帳號綁定來源；舊報名資料中的 ownerUid 會被前端忽略。
- 已綁定角色只有本人或管理者可修改。
- 未綁定角色可用角色名稱修改與新增報名。
- 管理者 exl78000@gmail.com 可修改資料，但不能把已被其他帳號綁定的角色改綁給自己。

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }
    function isAdmin() {
      return signedIn() && request.auth.token.email == 'exl78000@gmail.com';
    }
    function isUnlinked() {
      return resource.data.unlinkedAt != null && resource.data.unlinkedAt != '';
    }
    function isGoogleOwned() {
      return !isUnlinked() && resource.data.ownerUid != null && resource.data.ownerUid != '';
    }
    function isOwner() {
      return signedIn() && !isUnlinked() && resource.data.ownerUid == request.auth.uid;
    }
    function createUnboundOrSelfOwned() {
      return isAdmin()
        || (!('ownerUid' in request.resource.data) || request.resource.data.ownerUid == '')
        || (signedIn() && request.resource.data.ownerUid == request.auth.uid);
    }
    function adminSafeCharacterUpdate() {
      return isAdmin() && (
        !isGoogleOwned()
        || request.resource.data.ownerUid == resource.data.ownerUid
        || request.resource.data.ownerUid == ''
        || !('ownerUid' in request.resource.data)
      );
    }

    match /characters/{characterId} {
      allow read: if true;
      allow create: if createUnboundOrSelfOwned();
      allow update: if adminSafeCharacterUpdate() || isOwner() || !isGoogleOwned();
      allow delete: if isAdmin();
    }

    match /signups/{signupId} {
      allow read: if true;
      // 報名資料不綁定帳號；角色權限由前端依 characters 主檔判斷。
      // 若要做到後端強驗證，需要 Cloud Functions 或調整資料結構，把 ownerUid 複製到 signup 並同步維護。
      allow create, update, delete: if true;
    }

    match /users/{userId}/characters/{characterId} {
      allow read, write: if signedIn() && (request.auth.uid == userId || isAdmin());
    }

    match /adminEvents/{eventId} {
      allow read, write: if isAdmin();
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
